void helper(Node*root,vector<int>&ans){
    if(root==NULL){
        return;
    }
    
    ans.push_back(root->data);
    helper(root->left,ans);
    helper(root->right,ans);

}


class Solution {
  public:
    vector<int> preorder(Node* root) {
        // code here
        vector<int>ans;
        helper(root,ans);
        return ans;
    }
};
